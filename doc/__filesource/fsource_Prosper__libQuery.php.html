  311.   /**
  312.    * Factory function that creates a select statement query object.
  313.    * @param varargs [optional] If nothing is provided assumes 'select *'
  314.    *       otherwise processes each argument as a column,
  315.    *        array arguments are processed as any number of aliased columns where column_name => alias
  316.    * @return Select Statement Query Object
  317.    */
  318.   static function select() {
  319.     if(func_num_args() == 0) {
  320.       $columns = "*";
  321.     } else {
  322.       $args = func_get_args();
  323.       foreach($args as $arg) {
  324.         if(is_array($arg)) {
  325.           foreach($arg as $key => $value) {
  326.             $parts[] = self::column($key) . self::alias($value);
  327.           }
  328.         } else {
  329.           $parts[] = self::column($arg);
  330.         }
  331.       }
  332.       $columns = implode(', ', $parts);
  333.     }
  334.     return new Query("select $columns", self::SELECT_STMT);
  335.   }

  435.   /**
  436.    * Used by where and on to parse conditional strings, interpolating parameters,
  437.    * quoting, escaping, and cross-platform replacements where needed
  438.    * @param string $predicate result predicate, i.e. 'where' or 'on'
  439.    * @param string $clause conditional clause
  440.    */
  441.   function conditional($predicate, $clause, $args) {
  442.     $result = " $predicate";
  443.     $named = $args[count($args) - 1];    
  444.     while($token = Token::next($clause)) {
  445.       $result .= $this->process_token($clause, $token, $args, $named);
  446.     }
  447.     $this->sql .= $result;
  448.     return $this;
  449.   }

  457.    * @return tokenized string
  458.    */
  459.   function process_token(&$clause, $token, &$args, $named) {
  460.     switch($token['type']) {
  461.       case Token::BOOLEAN:
  462.         return " " . (strtolower($token['token']) == 'true' ? self::true_value() : self::false_value());
  463.         break;
  464.       case Token::LITERAL:
  465.         return " " . self::escape($token['token']);
  466.         break;
  467.       case Token::MODULUS:
  468.         $func_args = $this->parse_func_args($clause, $args, $named);
  469.         if(count($func_args) == 2) {
  470.           return self::modulus($func_args[0], $func_args[1]);
  471.         }
  472.         break;
  473.       case Token::NAMED_PARAM:
  474.         return self::parameterize($named[substr($token['token'], 1)]);
  475.         break;
  476.       case Token::PARAMETER:
  477.         return self::parameterize(array_shift($args));
  478.         break;
  479.       case Token::SQL_ENTITY:
  480.         return " " . self::quote($token['token']);
  481.         break;
  482.       default:
  483.         return " " . $token['token'];
  484.         break;
  485.     }
  486.   }

  503.   /**
  504.    * Parses sql function arguments into a php array
  505.    * @param $clause sql clause to process
  506.    * @param $args arguments used for interpolation
  507.    * @param $named named interpolation arguments
  508.    * @return array arguments
  509.    */
  510.   function parse_func_args(&$clause, &$args, $named) {
  511.     $token = Token::next($clause);    
  512.     if($token['type'] == Token::OPEN_PAREN) {
  513.       $depth = 1;
  514.       $temp = "";
  515.       while($depth > 0) {
  516.         $token = Token::next($clause);
  517.         switch($token['type']) {
  518.           case Token::OPEN_PAREN:
  519.             $depth += 1;
  520.             $temp .= $this->process_token($clause, $token, $args, $named);
  521.             break;
  522.           case Token::CLOSE_PAREN:
  523.             $depth -= 1;
  524.             if($depth == 0) {
  525.               $result[] = $temp;
  526.             }
  527.             $temp .= $this->process_token($clause, $token, $args, $named);
  528.             break;
  529.           case Token::COMMA:
  530.             if($depth == 1) {
  531.               $result[] = $temp;
  532.               $temp = "";
  533.             }
  534.             break;
  535.           default:
  536.             $temp .= $this->process_token($clause, $token, $args, $named);
  537.             break;
  538.         }
  539.       }
  540.       return $result;
  541.     } else {
  542.       //Something is wrong -- return the processed token
  543.       return $this->process_token($clause, $tkn, $args, $named);
  544.     }
  545.   }
