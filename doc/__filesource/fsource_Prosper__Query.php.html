  325.    * @param varargs [optional] If nothing is provided assumes 'select *'
  326.    *       otherwise processes each argument as a column,

  330.   static function select() {

  332.       $columns = "*";

  334.       $args = func_get_args();
  335.       foreach($args as $arg) {
  336.         if(is_array($arg)) {
  337.           foreach($arg as $key => $value) {
  338.             $parts[] = self::column($key) . self::alias($value);
  339.           }

  341.           $parts[] = self::column($arg);

  464.   /**
  465.    * Processes a token
  466.    * @param string $clause sql cluase to process, by reference
  467.    * @param array $token token array
  468.    * @param array $args arguments used for interpolation, by reference
  469.    * @param array $named associative array
  470.    * @return string processed clause
  471.    */
  472.   function process_token(&$clause, $token, &$args, $named) {
  473.     switch($token['type']) {
  474.       case Token::BOOLEAN:
  475.         return " " . (strtolower($token['token']) == 'true' ? self::true_value() : self::false_value());
  476.         break;
  477.       case Token::LITERAL:
  478.         return " " . self::escape($token['token']);
  479.         break;
  480.       case Token::MODULUS:
  481.         $func_args = $this->parse_func_args($clause, $args, $named);
  482.         if(count($func_args) == 2) {
  483.           return self::modulus($func_args[0], $func_args[1]);
  484.         }
  485.         break;
  486.       case Token::NAMED_PARAM:
  487.         return self::parameterize($named[substr($token['token'], 1)]);
  488.         break;
  489.       case Token::PARAMETER:
  490.         return self::parameterize(array_shift($args));
  491.         break;
  492.       case Token::SQL_ENTITY:
  493.         return " " . self::quote($token['token']);
  494.         break;
  495.       default:
  496.         return " " . $token['token'];
  497.         break;
  498.     }
  499.   }

  516.   /**
  517.    * Parses sql function arguments into a php array
  518.    * @param string $clause sql clause to process, by reference
  519.    * @param array $args arguments used for interpolation, by reference
  520.    * @param array $named named interpolation arguments
  521.    * @return array arguments
  522.    */
  523.   function parse_func_args(&$clause, &$args, $named) {
  524.     $token = Token::next($clause);    
  525.     if($token['type'] == Token::OPEN_PAREN) {
  526.       $depth = 1;
  527.       $temp = "";
  528.       while($depth > 0) {
  529.         $token = Token::next($clause);
  530.         switch($token['type']) {
  531.           case Token::OPEN_PAREN:
  532.             $depth += 1;
  533.             $temp .= $this->process_token($clause, $token, $args, $named);
  534.             break;
  535.           case Token::CLOSE_PAREN:
  536.             $depth -= 1;
  537.             if($depth == 0) {
  538.               $result[] = $temp;
  539.             }
  540.             $temp .= $this->process_token($clause, $token, $args, $named);
  541.             break;
  542.           case Token::COMMA:
  543.             if($depth == 1) {
  544.               $result[] = $temp;
  545.               $temp = "";
  546.             }
  547.             break;
  548.           default:
  549.             $temp .= $this->process_token($clause, $token, $args, $named);
  550.             break;
  551.         }
  552.       }
  553.       return $result;
  554.     } else {
  555.       //Something is wrong -- return the processed token
  556.       return $this->process_token($clause, $tkn, $args, $named);
  557.     }
  558.   }
